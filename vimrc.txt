" Indentation settings.
" Indentation settings for using 3 spaces instead of tabs
" Tab's default value isn't changed.
set shiftwidth=3
set softtabstop=3
set expandtab

" Use insensitive search, except when using capital letters
set ignorecase
set smartcase

"Allow backspacing over autoindent, line breaks and start of of insertion
"action
set backspace=indent,eol,start

"When opening a new line and no filetype-specific indenting is enabled, keeps
"the same indent as the line you're currently on.
set autoindent

"Display line numbers on the left
set number

highlight ExtraWhitespace ctermbg = white
"match ExtraWhitespace /\s\+$/
match ExtraWhitespace /\S\zs\s\+$/
